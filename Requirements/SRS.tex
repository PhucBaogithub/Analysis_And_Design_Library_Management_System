\chapter{Giới thiệu}
\section{Mục đích}
Tài liệu mô tả đầy đủ các yêu cầu nghiệp vụ, chức năng, dữ liệu, giao diện và chất lượng của Hệ thống quản lý thư viện. Mỗi yêu cầu được gắn mã duy nhất để làm căn cứ thiết kế, lập trình, kiểm thử chấp nhận người dùng (UAT) và quản lý thay đổi.

\section{Phạm vi sản phẩm}
Hệ thống số hóa việc quản lý tài khoản độc giả, thẻ thư viện, thể loại, đầu sách, bản sao sách, đăng ký mượn, phiếu mượn và phiếu trả. Độc giả tự phục vụ việc đăng ký, tra cứu và theo dõi lịch sử; thủ thư quản trị dữ liệu và xử lý vòng đời mượn--trả.

\subsection{Trong phạm vi}
\begin{itemize}
\item Xác thực, phân quyền hai vai trò Độc giả và Thủ thư.
\item Quản lý danh mục, đầu sách và từng bản sao vật lý.
\item Đăng ký thẻ, đăng ký mượn, lập/cập nhật phiếu mượn và phiếu trả.
\item Tra cứu, lịch sử giao dịch, phí phạt và báo cáo vận hành cơ bản.
\item Nhật ký các thao tác quản trị quan trọng và bảo vệ dữ liệu cá nhân.
\end{itemize}
\subsection{Ngoài phạm vi phiên bản 1.0}
Mua sắm sách, kế toán/thu tiền trực tuyến, RFID/cổng từ, liên thông thư viện khác, ứng dụng di động native, sách điện tử có DRM và gợi ý sách bằng AI không thuộc baseline này.

\section{Giả định và ràng buộc}
\begin{itemize}
\item Thư viện có kết nối mạng nội bộ/Internet ổn định; mỗi người dùng có một tài khoản duy nhất.
\item Độc giả từ đủ 16 tuổi theo report gốc; chính sách có thể cấu hình khi đơn vị nghiệp vụ xác nhận.
\item Phiên bản đầu là ứng dụng web responsive, ngôn ngữ tiếng Việt và múi giờ Việt Nam (UTC+7).
\item Hệ quản trị CSDL quan hệ phải hỗ trợ giao dịch ACID, khóa ngoại và sao lưu.
\item Thời hạn mượn, số sách tối đa, hạng thẻ và công thức phạt là tham số cấu hình, không ghi cố định trong mã nguồn.
\end{itemize}

\section{Quy tắc dữ liệu}
Khóa chính không tái sử dụng; khóa ngoại bảo đảm toàn vẹn tham chiếu; thời điểm lưu kèm múi giờ; tiền phạt dùng kiểu decimal; chuỗi tìm kiếm được chuẩn hóa Unicode. Dữ liệu giao dịch và audit không xóa cứng. Mật khẩu chỉ tồn tại ở dạng băm. Thời hạn lưu PII và audit phải được đại diện thư viện phê duyệt trước go-live.

\section{Giao diện người dùng}
\begin{itemize}
\item Màn hình công khai: trang chủ, tra cứu, chi tiết đầu sách, đăng ký và đăng nhập.
\item Khu vực độc giả: hồ sơ, thẻ, đăng ký mượn, lịch sử và cảnh báo.
\item Khu vực thủ thư: dashboard, tài khoản/thẻ, danh mục/sách, mượn/trả, báo cáo và audit.
\item Mọi biểu mẫu hiển thị trường bắt buộc, lỗi cạnh trường, xác nhận thao tác nguy hiểm và phản hồi thành công/thất bại.
\end{itemize}

\section{Giao diện phần mềm và truyền thông}
Phiên bản 1.0 cung cấp giao diện web qua HTTPS và API nội bộ dạng JSON/REST (hoặc tương đương) có versioning. CSDL không được truy cập trực tiếp từ trình duyệt. Dịch vụ email/SMS, cổng thanh toán và RFID chưa phải phụ thuộc bắt buộc; khi bổ sung phải qua change request và đặc tả hợp đồng tích hợp.

\chapter{Use case trọng yếu và tiêu chí chấp nhận}
\section{UC-12 -- Đăng nhập}
\textbf{Actor:} Độc giả/Thủ thư. \textbf{Tiền điều kiện:} tài khoản hoạt động. \textbf{Kích hoạt:} người dùng gửi thông tin đăng nhập.
\begin{enumerate}
\item Hệ thống kiểm tra định dạng, giới hạn tần suất và đối chiếu mật khẩu băm.
\item Nếu hợp lệ, hệ thống tạo phiên, nạp vai trò và chuyển đến trang phù hợp.
\item Nếu không hợp lệ, hệ thống thông báo chung và tăng bộ đếm thất bại.
\end{enumerate}
\textbf{Chấp nhận:} đúng thông tin thì vào đúng vai trò; sai thông tin không tạo phiên; lần sai thứ 5 trong cửa sổ 15 phút kích hoạt khóa tạm và được ghi log.

\section{UC-22 -- Đăng ký mượn sách}
\textbf{Tiền điều kiện:} độc giả đã đăng nhập, thẻ hiệu lực. \textbf{Luồng chính:} chọn sách sẵn sàng; hệ thống kiểm tra thẻ/hạn mức/quá hạn/tồn kho; độc giả xác nhận; hệ thống tạo yêu cầu/phiếu và giữ trạng thái nhất quán. \textbf{Ngoại lệ:} thẻ không hợp lệ, hết sách, vượt hạn mức hoặc đang quá hạn thì từ chối và nêu lý do có thể khắc phục.

\section{UC-RETURN -- Ghi nhận trả sách}
\textbf{Actor:} Thủ thư. Thủ thư tìm phiếu đang mở, quét/chọn bản sao, nhập tình trạng và xác nhận. Hệ thống tính đề xuất phí phạt, đóng chi tiết mượn, ghi phiếu trả và cập nhật bản sao trong một giao dịch. Trả trùng hoặc sách không thuộc phiếu phải bị từ chối.
